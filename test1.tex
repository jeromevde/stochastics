\section{Vasicek short rate model}

\begin{center}
    \begin{tcolorbox}[
        colback=white, 
        colframe=gray!75!black, 
        boxrule=1pt,
        halign=left,
        valign=top,
        sharp corners,
        boxsep=10pt,
        left=10pt, right=10pt, top=10pt, bottom=10pt
    ]
        We refer to [BM06] for a detailed analysis of Vasicek’s model or to the original paper of Vasicek [Vas76].

        Vasicek’s model is a standard choice to model interest rates. It is expressed in terms of the short rate:
        \[
        r(t) = \lim_{dt\to0} R(t,t+ dt)
        \]
        where $R(t,T)$ denotes the continuously compounded zero coupon rate for maturity $T$ at time $t$. The model is then defined by the SDE:
        \[
        dr(t) = \kappa(\theta-r(t))dt+ \sigma dB(t)
        \]
        where $B(t)$ denotes standard Brownian motion.

        The model has the following features:
        \begin{itemize}
            \item Analytic formulas for bond prices and bond option prices in terms of short rate.
            \item Exact simulation of short rate and integrated short rate possible.
        \end{itemize}
    \end{tcolorbox}
\end{center}

\subsection{Solving the Vasicek SDE}

The Vasicek model is an example of an Ornstein-Uhlenbeck process. We can find an explicit formula for $r(t)$ by solving the SDE.

\begin{enumerate}
    \item \textbf{Rearrange the SDE:}
    \begin{align*}
        dr(t) &= \kappa(\theta - r(t))dt + \sigma dB(t) \\
        dr(t) + \kappa r(t) dt &= \kappa \theta dt + \sigma dB(t)
    \end{align*}

    \item \textbf{Introduce an Integrating Factor:} We use the integrating factor $e^{\kappa t}$. Let's consider the process $X_t = e^{\kappa t} r(t)$. We can find its differential using Itô's Lemma.
    \[
    d(e^{\kappa t} r(t)) = \kappa e^{\kappa t} r(t) dt + e^{\kappa t} dr(t)
    \]

    \item \textbf{Substitute the SDE:} Now, substitute the expression for $dr(t)$ from step 1 into this equation:
    \[
    d(e^{\kappa t} r(t)) = \kappa e^{\kappa t} r(t) dt + e^{\kappa t} (\kappa \theta dt - \kappa r(t) dt + \sigma dB(t))
    \]

    \item \textbf{Simplify:} The terms involving $r(t)dt$ cancel out, leaving a much simpler differential:
    \[
    d(e^{\kappa t} r(t)) = \kappa \theta e^{\kappa t} dt + \sigma e^{\kappa t} dB(t)
    \]

    \item \textbf{Integrate:} We can now integrate both sides from time 0 to $t$:
    \begin{align*}
        \int_0^t d(e^{\kappa s} r(s)) &= \int_0^t \kappa \theta e^{\kappa s} ds + \int_0^t \sigma e^{\kappa s} dB(s) \\
        e^{\kappa t} r(t) - r(0) &= \kappa \theta \left[\frac{1}{\kappa}e^{\kappa s}\right]_0^t + \sigma \int_0^t e^{\kappa s} dB(s) \\
        e^{\kappa t} r(t) - r(0) &= \theta (e^{\kappa t} - 1) + \sigma \int_0^t e^{\kappa s} dB(s)
    \end{align*}

    \item \textbf{Solve for r(t):} Finally, multiply by $e^{-\kappa t}$ to isolate $r(t)$:
    \[
    r(t) = r(0)e^{-\kappa t} + \theta(1 - e^{-\kappa t}) + \sigma \int_0^t e^{-\kappa(t-s)} dB(s)
    \]
\end{enumerate}

This is the explicit solution for the Vasicek short rate model. It shows that $r(t)$ is normally distributed, which is a key feature that allows for analytic formulas for bond prices.
